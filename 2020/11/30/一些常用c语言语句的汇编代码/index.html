不同C语言语句在汇编中的显示情况及简单分析

1.if-else

源代码如下

#include<stdio.h>

int main(void)
{
    int a=1,b=2,c;
    if(a<b)
    {
        c=0;
    }
    else
    {
        c=1;
    }
    return 0;
}

for循环

简单的累加代码

#include<stdio.h>

int main(void)
{
    int i,sum;
    sum=0;

    for(i=0;i<5;i++)
    {
        sum=i+sum;
    }

    printf("%d",sum);
}

while循环

#include<stdio.h>

int main(void)
{
    int i=0;
    int sum=0;
    while(i<5)
    {
        sum=sum+i;
        i++;
    }
    printf("%d",sum);
}

do_while的汇编语句

#include<stdio.h>

int main(void)
{
    int i=0;
    int sum=0;

    do
    {
        sum=sum+i;
        i++;
    }
    while(i<5);
    printf("%d",sum);
}
