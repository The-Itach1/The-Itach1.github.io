猜数游戏进阶版2

print('...........猜数游戏............')
temp=input("写出你猜的数：")#相对于scanf
guess=int(temp)#将字符串转化为int型
while guess!=8:#加了一个while循环
    if guess==8:#注意:号，并且下次按enter会自动空出一个tab
        print("you are right")
else:
    if guess<8:
        print("输入的数小了")
    else:
        print("输入的数大了")
    temp=input("写出你猜的数：")
    guess=int(temp)#将字符串转化为int型
print("you are right")
print("game over")

Python的一些运算符

+,-,*,/,**,//

and，or，&，|

# 判断变量是否为0， 是0则为False，非0判断为True，
# and中含0，返回0； 均为非0时，返回后一个值， 
2 and 0   # 返回0
2 and 1   # 返回1
1 and 2   # 返回2

# or中， 至少有一个非0时，返回第一个非0,
2 or 0   # 返回2
2 or 1   # 返回2
0 or 1   # 返回1 


# 1&2，2在二进制里面是10,1在二进制中是01，那么01与运算10得到是0 
1 & 2 # 输出为 0， 
1 | 2  # 输出为3

向列表中添加元素

append()

>>> menber=['i','1','我爱你',[1,2,3]]
>>> menber.append ('不对劲')#只能加一个元素
>>> menber
['i', '1', '我爱你', [1, 2, 3], '不对劲']

extend()

>>> menber=['i', '1', '我爱你', [1, 2, 3], '不对劲']
>>> menber.extend(['还可以','问题不大'])#注意写法
>>> menber
['i', '1', '我爱你', [1, 2, 3], '不对劲', '还可以', '问题不大']

insert()

>>> menber=['i', '1', '我爱你', [1, 2, 3], '不对劲']
>>> menber.insert(1,'有问题')#1是指插入的位置，后面是内容。但数组是从0开始排的，所以有问题在下面的第二个。
>>> menber
['i', '有问题', '1', '我爱你', [1, 2, 3], '不对劲']

向列表中删除一些元素

remove()

>>> menber=['i', '1', '我爱你', [1, 2, 3], '不对劲', '还可以', '问题不大']
>>> menber
['i', '1', '我爱你', [1, 2, 3], '不对劲', '还可以', '问题不大']
>>> menber.remove('我爱你')#知道元素内容，并删除这个元素，若元素中有两个相同的元素，那么只会删除第一个。
>>> menber
['i', '1', [1, 2, 3], '不对劲', '还可以', '问题不大']

del list[] 注意括号是[],里面写删去元素所在位置，从0开始数

>>> menber=['i', '1', '我爱你', [1, 2, 3], '不对劲', '还可以', '问题不大']
>>> del menber[0]
>>> menber
['1', '我爱你', [1, 2, 3], '不对劲', '还可以', '问题不大']

pop() 指汇编中的出栈pop，将最后一个元素出栈。

>>> menber=['i', '1', '我爱你', [1, 2, 3], '不对劲', '还可以', '问题不大']
>>> menber
['i', '1', '我爱你', [1, 2, 3], '不对劲', '还可以', '问题不大']
>>> menber.pop()#这样默认出栈最后一个元素。
'问题不大'
>>> menber
['i', '1', '我爱你', [1, 2, 3], '不对劲', '还可以']
>>> menber.pop(2)#这样指出栈指定元素，从0开始计数，和remove用法差不多。
'我爱你'
>>> menber
['i', '1', [1, 2, 3], '不对劲', '还可以']

列表分片

列表分片，将列表拷贝，可分成多个列表，以方便多个元素操作，并且不会改变原列表的元素。

>>> menber=['i', '1', '我爱你', [1, 2, 3], '不对劲', '还可以', '问题不大']
>>> menber
['i', '1', '我爱你', [1, 2, 3], '不对劲', '还可以', '问题不大']
>>> menber[1:5]
['1', '我爱你', [1, 2, 3], '不对劲']
>>> menber[:3]
['i', '1', '我爱你']
>>> menber[2:]
['我爱你', [1, 2, 3], '不对劲', '还可以', '问题不大']
>>> menber[:]#可用来拷贝一份原列表，对其进行操作，就可以实现不改变原列表内容。
['i', '1', '我爱你', [1, 2, 3], '不对劲', '还可以', '问题不大']
>>> menber2=menber[:]
>>> menber2
['i', '1', '我爱你', [1, 2, 3], '不对劲', '还可以', '问题不大']
